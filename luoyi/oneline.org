** BASH read
打开特定文件，一行一行读入

#+begin_src sh
exec 3< "MyFile"
 
while read -u 3; do 
    A=$(echo $REPLY |perl -F'\s+' -ane 'print  $F[0]') 
    B=$(echo $REPLY |perl -F'\s+' -ane 'print  $F[1]') 
    echo -e "$A\t$B"
done  
 
exec 3>&- 
#+end_src

** Perl Read 
一次读入整个文件

#+begin_src perl
#!/usr/bin/env perl
use strict;
use utf8;

binmode(STDIN, ":utf8");
binmode(STDOUT, ":utf8");

my $content = '';
open my $fh, "<:encoding(UTF-8)", "$ARGV[0]" or die $!;
{
	local $/;
	$content = <$fh>;
}
close $fh;
#+end_src
